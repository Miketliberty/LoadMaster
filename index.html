// JavaScript for calculating total cargo weight and validating weight restriction const cargoRows = document.querySelectorAll('.cargoRow'); const totalCargoWeightElement = document.getElementById('totalCargoWeight'); const weightErrorElement = document.getElementById('weightError'); function calculateTotalCargoWeight() { let totalCargoWeight = 0; cargoRows.forEach(row => { const weightInput = row.querySelector('.cargoWeight'); totalCargoWeight += parseFloat(weightInput.value) || 0; }); totalCargoWeightElement.textContent = totalCargoWeight.toFixed(2); return totalCargoWeight; } function validateWeightRestriction() { const totalCargoWeight = calculateTotalCargoWeight(); if (totalCargoWeight > 21975) { weightErrorElement.textContent = 'Total weight exceeds 21975 kg'; } else { weightErrorElement.textContent = 'Load within CG limits'; } } } cargoRows.forEach(row => { const weightInput = row.querySelector('.cargoWeight'); weightInput.addEventListener('input', () => { validateWeightRestriction(); }); });
